local function GetTarget()
    local world = workspace:FindFirstChild("World Mobs")
    if not world then return nil end

    local bossFolder = workspace.Misc.MobSpawnInfo
    local mobFolder = world:FindFirstChild("Mobs")
    local worldmobsfolder = workspace["World Mobs"]["Event Mobs"]
    -- ƯU TIÊN BOSS
    if bossFolder then
        for _, v in pairs(bossFolder:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                return v
            end
        end
    end

    -- NẾU KHÔNG CÓ BOSS → FARM MOB THƯỜNG
    if mobFolder then
        for _, v in pairs(mobFolder:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                return v
            end
        end
    end

    return nil
end

-- Hàm lấy CFrame động theo người chơi
local function GetPlayerCFrame()
    local char = game.Players.LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        return char.HumanoidRootPart.CFrame
    end
    return CFrame.new(0, 100, 0) -- fallback
end

-- Tạo args cho skill
local function CreateArgs(skillId, began)
    local rootCFrame = GetPlayerCFrame()
    
    -- Điều chỉnh offset tùy game của bạn (dùng giá trị gốc bạn cung cấp làm base)
    local cameraCf = rootCFrame * CFrame.new(3.6, 2.8, 11.7)   -- approx từ ví dụ của bạn
    local playerCf  = rootCFrame * CFrame.new(-3.6, -2.8, 11.7)
    
    -- Aim: tạm hướng về phía trước, bạn có thể chỉnh thành lock enemy nếu cần
    local aimPos = rootCFrame.Position + rootCFrame.LookVector * 40
    
    return {
        {
            Camera = cameraCf,
            SkillId = skillId,
            Began = began,
            CFrame = playerCf,
            ["Typ\208\181"] = 1,
            Aim = Vector3.new(aimPos.X, aimPos.Y, aimPos.Z)
        }
    }
end

-- Hàm spam skill (chạy trong spawn)
local function StartAuto()
    local skillId = SkillIds[SelectedMode]
    if not skillId then return end
    
    spawn(function()
        while AutoEnabled do
            local args = CreateArgs(skillId, true)
            pcall(function()
                game:GetService("ReplicatedStorage").Remotes.SkillRemote:FireServer(unpack(args))
            end)
            wait(0.05)
        end
        
        -- Khi tắt auto → gửi Began = false
        local argsOff = CreateArgs(skillId, false)
        pcall(function()
            game:GetService("ReplicatedStorage").Remotes.SkillRemote:FireServer(unpack(argsOff))
        end)
    end)
end

local dropdown
local playerList = {}

local function refreshPlayers()
	playerList = {}
	for _,v in pairs(Players:GetPlayers()) do
		table.insert(playerList, v.Name)
	end

	if dropdown then
		dropdown:Refresh(playerList, true)
	end
end

dropdown = Tab2:AddDropdown({
	Name = "Chọn Player",
	Default = "",
	Options = playerList,
	Callback = function(selected)

		local charFolder = workspace:FindFirstChild("Characters")
		if not charFolder then return end

		local target = charFolder:FindFirstChild(selected)
		if not target then return end

		local status = target:FindFirstChild("Status")
		if not status then return end

		local info = ""

		for _,v in pairs(status:GetChildren()) do
			pcall(function()
				if v:IsA("BoolValue") then
					v.Value = true
					info = info .. v.Name .. " = true\n"
				elseif v:IsA("NumberValue") then
					info = info .. v.Name .. " = " .. v.Value .. "\n"
				end
			end)
		end

		OrionLib:MakeNotification({
			Name = "Status "..selected,
			Content = info ~= "" and info or "Không có dữ liệu",
			Time = 10
		})
	end
})

local Section = Tab2:AddSection({
	Name = "Check time boss"
})

local bossLabels = {}

local function createLabel(name, text)
	if bossLabels[name] then return end
	bossLabels[name] = Section:AddLabel(name.." : "..text)
end

local function updateLabel(name, text)
	if bossLabels[name] then
		bossLabels[name]:Set(name.." : "..text)
	end
end

local function removeLabel(name)
	if bossLabels[name] then
		bossLabels[name]:Destroy()
		bossLabels[name] = nil
	end
end

task.spawn(function()
	while true do
		pcall(function()
			local misc = workspace:FindFirstChild("Misc")
			local mobFolder = misc and misc:FindFirstChild("MobSpawnInfo")

			if mobFolder then
				for _,mob in pairs(mobFolder:GetChildren()) do
					local timer = mob:FindFirstChild("MobRespawnTimer")

					if timer and timer:FindFirstChild("TextLabel") then
						local text = timer.TextLabel.Text

						if text ~= "00" then
							if not bossLabels[mob.Name] then
								createLabel(mob.Name, text)
							else
								updateLabel(mob.Name, text)
							end
						else
							removeLabel(mob.Name)
						end
					end
				end
			end
		end)

		task.wait(1)
	end
end)

					table.insert(mobs, hrp)
				end
			end
		end
	end

	return mobs
end

								if hrp then
									table.insert(mobs, hrp)
								end
							end
						end
					end

					if #mobs > 0 then
						if currentIndex > #mobs then
							currentIndex = 1
						end

						if tick() - lastSwitch >= 3 then
							currentIndex += 1

							if currentIndex > #mobs then
								currentIndex = 1
							end

							lastSwitch = tick()
						end

						local mob = mobs[currentIndex]

						if mob then
							local pos = mob.Position

							local args = {
								{
									Camera = CFrame.new(
										pos + Vector3.new(0,3,8),
										pos
									),

									SkillId = "101",
									Began = true,

									CFrame = CFrame.new(
										pos + Vector3.new(0,0,5),
										pos
									),

									["Typ\208\181"] = 1,

									Aim = vector.create(
										pos.X,
										pos.Y,
										pos.Z
									)
								}
							}

							game:GetService("ReplicatedStorage")
								.Remotes.SkillRemote
								:FireServer(unpack(args))
						end
					end

					task.wait(0.1)
				end
			end
		end)
	end
})
